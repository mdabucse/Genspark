4.2 JWT Token Helper
Helpers/JwtHelper.cs
public string GenerateToken(User user) {
    var claims = new[] {
        new Claim(ClaimTypes.NameIdentifier, user.Id.ToString()),
        new Claim(ClaimTypes.Email, user.Email),
        new Claim(ClaimTypes.Role, user.Role)
    };
    var key = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_config["JwtSettings:Secret"]!));
    var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
    var token = new JwtSecurityToken(
        issuer: _config["JwtSettings:Issuer"],
        audience: _config["JwtSettings:Audience"],
        claims: claims,
        expires: DateTime.UtcNow.AddDays(7),
        signingCredentials: creds);
    return new JwtSecurityTokenHandler().WriteToken(token);
}

4.3 Auth Service (Register + Login Logic)
Services/AuthService.cs — Key methods
// Register
public async Task<AuthResponseDto> RegisterAsync(RegisterDto dto) {
    if (await _db.Users.AnyAsync(u => u.Email == dto.Email))
        throw new Exception("Email already registered");
    var user = new User {
        Name = dto.Name, Email = dto.Email, Phone = dto.Phone,
        Role = dto.Role,
        PasswordHash = BCrypt.Net.BCrypt.HashPassword(dto.Password)
    };
    _db.Users.Add(user); await _db.SaveChangesAsync();
    return new AuthResponseDto { Token = _jwt.GenerateToken(user), Role = user.Role };
}

// Login
public async Task<AuthResponseDto> LoginAsync(LoginDto dto) {
    var user = await _db.Users.FirstOrDefaultAsync(u => u.Email == dto.Email);
    if (user == null || !BCrypt.Net.BCrypt.Verify(dto.Password, user.PasswordHash))
        throw new Exception("Invalid credentials");
    return new AuthResponseDto { Token = _jwt.GenerateToken(user), Role = user.Role };
}

5.6 Trip Search Logic — Key Algorithm
TripService.cs — SearchTripsAsync
public async Task<List<TripDto>> SearchTripsAsync(string source, string dest, DateTime date) {
    return await _db.Trips
        .Include(t => t.Bus)
        .Include(t => t.Route)
        .Where(t =>
            t.Route.Source.ToLower() == source.ToLower() &&
            t.Route.Destination.ToLower() == dest.ToLower() &&
            t.DepartureTime.Date == date.Date &&
            t.Status == "scheduled")
        .Select(t => new TripDto {
            TripId = t.Id,
            BusName = t.Bus.BusName,
            BusType = t.Bus.BusType,
            DepartureTime = t.DepartureTime,
            ArrivalTime = t.ArrivalTime,
            TotalFare = t.BaseFare + (t.BaseFare * t.TaxPercent / 100),
            AvailableSeats = t.Bus.Capacity - _db.TripSeatStatuses
                .Count(ss => ss.TripId == t.Id && ss.Status == "booked")
        }).ToListAsync();
}


STEP 6
Phase 6 — Seat Management & Atomic Seat Locking

WHY Seat Locking is Critical
Problem: Two users select the same seat simultaneously → both pay → double booking (revenue & trust loss).
Solution: When a user selects a seat, IMMEDIATELY lock it for 5 minutes in the database (or Redis).
If payment is not completed within 5 minutes, the lock expires and the seat becomes available again.
This lock must be ATOMIC — use a database transaction to prevent race conditions.

6.1 Seat Lock Flow (Step by Step)
User calls POST /api/seats/lock with { tripId, seatIds[], userId }
Backend begins a DB transaction
For EACH seat, check TripSeatStatus: is status = 'available'?
If ANY seat is not available → rollback transaction → return error
If ALL seats available → set status = 'locked', locked_by = userId, locked_until = NOW() + 5 min
Commit transaction → return success with lock expiry time
Frontend shows 5-minute countdown timer
Background job (or check-on-use): if locked_until < NOW() → reset to 'available'

6.2 Seat Lock Service — Transaction Code
Services/SeatService.cs — LockSeatsAsync
public async Task<bool> LockSeatsAsync(int tripId, int[] seatIds, int userId) {
    using var tx = await _db.Database.BeginTransactionAsync(
        IsolationLevel.Serializable);  // ← IMPORTANT for atomicity
    try {
        var statuses = await _db.TripSeatStatuses
            .Where(s => s.TripId == tripId && seatIds.Contains(s.SeatId))
            .ToListAsync();

        // Check all must be available
        if (statuses.Any(s => s.Status != "available"))
            throw new Exception("One or more seats are no longer available");

        foreach (var seat in statuses) {
            seat.Status = "locked";
            seat.LockedBy = userId;
            seat.LockedUntil = DateTime.UtcNow.AddMinutes(5);
        }
        await _db.SaveChangesAsync();
        await tx.CommitAsync();
        return true;
    } catch { await tx.RollbackAsync(); throw; }
}

6.3 Expired Lock Cleanup (Background Service)
BackgroundServices/SeatUnlockJob.cs — runs every minute
protected override async Task ExecuteAsync(CancellationToken stoppingToken) {
    while (!stoppingToken.IsCancellationRequested) {
        await Task.Delay(TimeSpan.FromMinutes(1), stoppingToken);
        var expired = await _db.TripSeatStatuses
            .Where(s => s.Status == "locked" && s.LockedUntil < DateTime.UtcNow)
            .ToListAsync();
        expired.ForEach(s => { s.Status = "available"; s.LockedBy = null; s.LockedUntil = null; });
        await _db.SaveChangesAsync();
    }
}

7.3 Booking Service Logic

public async Task<BookingResponseDto> CreateBookingAsync(CreateBookingDto dto, int userId) {
    // 1. Verify all seats are still locked by THIS user
    var seatIds = dto.Passengers.Select(p => p.SeatId).ToArray();
    var locks = await _db.TripSeatStatuses
        .Where(s => s.TripId == dto.TripId && seatIds.Contains(s.SeatId))
        .ToListAsync();
    if (locks.Any(l => l.Status != "locked" || l.LockedBy != userId))
        throw new Exception("Seat lock expired or invalid");

    // 2. Calculate fare
    var trip = await _db.Trips.FindAsync(dto.TripId);
    var totalFare = trip!.BaseFare * dto.Passengers.Count;
    totalFare += totalFare * trip.TaxPercent / 100;

    // 3. Create booking (status = pending)
    var booking = new Booking {
        UserId = userId, TripId = dto.TripId,
        BookingRef = GenerateRef(),
        TotalAmount = totalFare, Status = "pending"
    };
    _db.Bookings.Add(booking); await _db.SaveChangesAsync();

    // 4. Add passengers
    foreach (var p in dto.Passengers) { /* add BookingPassenger rows */ }
    await _db.SaveChangesAsync();

    return new BookingResponseDto { BookingId = booking.Id, BookingRef = booking.BookingRef, TotalFare = totalFare };
}

7.4 Cancellation Logic
Services/BookingService.cs — CancelBookingAsync
public async Task CancelBookingAsync(int bookingId, int userId) {
    var booking = await _db.Bookings.Include(b => b.Trip)
        .FirstOrDefaultAsync(b => b.Id == bookingId && b.UserId == userId);

    if (booking == null) throw new Exception("Booking not found");
    if (booking.Status != "confirmed") throw new Exception("Cannot cancel");

    // Rule: Must be 24+ hours before departure
    var hoursLeft = (booking.Trip.DepartureTime - DateTime.UtcNow).TotalHours;
    if (hoursLeft < 24) throw new Exception("Cancellation window closed (< 24 hours)");

    booking.Status = "cancelled";
    booking.CancelledAt = DateTime.UtcNow;

    // Release seats back to available
    var seatStatuses = await _db.TripSeatStatuses
        .Where(s => s.TripId == booking.TripId &&
            _db.BookingPassengers.Where(bp => bp.BookingId == bookingId)
                .Select(bp => bp.SeatId).Contains(s.SeatId))
        .ToListAsync();
    seatStatuses.ForEach(s => { s.Status = "available"; s.LockedBy = null; });
    await _db.SaveChangesAsync();
    // Trigger refund process here
}

    // Mark all seats as booked
    var passengers = await _db.BookingPassengers
        .Where(p => p.BookingId == bookingId).ToListAsync();
    var seatIds = passengers.Select(p => p.SeatId).ToArray();
    var seats = await _db.TripSeatStatuses
        .Where(s => s.TripId == booking.TripId && seatIds.Contains(s.SeatId)).ToListAsync();
    seats.ForEach(s => { s.Status = "booked"; s.LockedBy = null; s.LockedUntil = null; });
    await _db.SaveChangesAsync();

    await _emailService.SendBookingConfirmationAsync(booking);
    return new PaymentResponseDto { Success = true, BookingRef = booking.BookingRef };
}

public async Task SendBookingConfirmationAsync(Booking booking) {
    var message = new MimeMessage();
    message.From.Add(new MailboxAddress("Bus Booking", _config["SmtpSettings:Username"]));
    message.To.Add(new MailboxAddress(booking.User.Name, booking.User.Email));
    message.Subject = $"Booking Confirmed — {booking.BookingRef}";

    var builder = new BodyBuilder();
    builder.HtmlBody = BuildConfirmationEmailHtml(booking);
    message.Body = builder.ToMessageBody();

    using var client = new SmtpClient();
    await client.ConnectAsync(_config["SmtpSettings:Host"],
        int.Parse(_config["SmtpSettings:Port"]!), SecureSocketOptions.StartTls);
    await client.AuthenticateAsync(
        _config["SmtpSettings:Username"], _config["SmtpSettings:Password"]);
    await client.SendAsync(message);
    await client.DisconnectAsync(true);
}

export const AuthGuard: CanActivateFn = () => {
  const router = inject(Router);
  const token = localStorage.getItem('token');
  if (!token) {
    router.navigate(['/login']);
    return false;
  }
  return true;
};

14.3 Role Guard
core/guards/role.guard.ts
export const RoleGuard: CanActivateFn = (route) => {
  const authService = inject(AuthService);
  const router = inject(Router);
  const requiredRole = route.data['role'];
  const userRole = authService.getCurrentUserRole();
  if (userRole !== requiredRole) {
    router.navigate(['/']);
    return false;
  }
  return true;
};

ngOnInit() {
  this.lockExpiryTime = new Date(Date.now() + 5 * 60 * 1000);
  this.interval = setInterval(() => {
    const secondsLeft = Math.floor((this.lockExpiryTime.getTime() - Date.now()) / 1000);
    if (secondsLeft <= 0) {
      clearInterval(this.interval);
      this.showTimeoutModal();  // seats released

    }

Phase 21 — Edge Cases & Error Handling

21.1 Critical Edge Cases — Backend
Edge Case
How to Handle
Double booking same seat
Atomic DB transaction with Serializable isolation (Phase 6)
Seat lock expires during payment
On payment confirm, re-verify seats are still locked by user; if not → fail payment, release order
Payment fails after seat lock
Background job releases locks after 5 min; refund if partial payment
Operator adds duplicate bus number
UNIQUE constraint on bus_number column — API returns 400 with clear message
Session timeout
JWT expiry; frontend interceptor catches 401, clears token, redirects to login
Cancellation window check
Calculate hours to departure before allowing cancel; return 400 if < 24 hrs
Admin blocks an operator
Set operator.is_active = false; all their future trips become unvisible in search
Trip cancelled by operator
Set trip.status = 'cancelled'; notify all passengers via email; auto-refund

21.2 Global Error Handling Middleware (.NET)
Middleware/ExceptionMiddleware.cs
app.UseExceptionHandler(builder => builder.Run(async ctx => {
    var exception = ctx.Features.Get<IExceptionHandlerFeature>()?.Error;
    ctx.Response.StatusCode = exception switch {
        UnauthorizedAccessException => 401,
        KeyNotFoundException => 404,
        ArgumentException => 400,
        _ => 500
    };
    ctx.Response.ContentType = "application/json";
    await ctx.Response.WriteAsJsonAsync(new {
        error = exception?.Message ?? "Internal Server Error"
    });
}));

21.3 Angular Error Interceptor
core/interceptors/error.interceptor.ts
export const errorInterceptor: HttpInterceptorFn = (req, next) => {
  return next(req).pipe(
    catchError((err: HttpErrorResponse) => {
      if (err.status === 401) {
        localStorage.removeItem('token');
        inject(Router).navigate(['/login']);
      }
      const message = err.error?.error || 'Something went wrong';
      inject(ToastrService).error(message);
      return throwError(() => err);
    })
  );
};
